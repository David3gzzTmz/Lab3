Codigo del Generador de Numeros

Acerca del código

Este programa tiene una sintaxis muy similar a la de C/C++; El programa utiliza las librerias de stdio.h y hardware/gpio.h También es preferible hacer las funciones voids.


Pasos

  1. Incluye librerias
  2. Define las variables
  3. Programa el setup de los pines utilizando _gpio_init y gpio_set_dir
  4. Crea los voids para definir presequencias de segmentos
  5. Programa la sequencia en el void loop

Código

#include #include "hardware/gpio.h"
#define LEDA 17
#define LEDB 16
#define LEDC 13
#define LEDD 15
#define LEDE 14
#define LEDF 18
#define LEDG 19
#define LEDP 12

void setup() { // put your setup code here, to run once: Serial1.begin(115200); Serial1.println("Hello, Raspberry Pi Pico!");
_gpio_init(LEDA); gpio_set_dir(LEDA, GPIO_OUT); _gpio_init(LEDB); gpio_set_dir(LEDB, GPIO_OUT); _gpio_init(LEDC); gpio_set_dir(LEDC, GPIO_OUT); _gpio_init(LEDD); gpio_set_dir(LEDD, GPIO_OUT); _gpio_init(LEDE); gpio_set_dir(LEDE, GPIO_OUT); _gpio_init(LEDF); gpio_set_dir(LEDF, GPIO_OUT); _gpio_init(LEDG); gpio_set_dir(LEDG, GPIO_OUT); _gpio_init(LEDP); }

void N8(int) { gpio_put(LEDC,0); gpio_put(LEDB,0); gpio_put(LEDA,0); gpio_put(LEDD,0); gpio_put(LEDE,0); gpio_put(LEDF,0); gpio_put(LEDG,0); }
void N1(int) { gpio_put(LEDB,0); gpio_put(LEDA,1); gpio_put(LEDD,1); gpio_put(LEDE,1); gpio_put(LEDF,1); gpio_put(LEDG,1); }
void N23E(int) { gpio_put(LEDA,0); gpio_put(LEDD,0); gpio_put(LEDG,0); }

void loop() { N8(1); gpio_put(LEDG,1); delay(1000);
N1(1); delay(1000);
N23E(1); gpio_put(LEDE,0); gpio_put(LEDC,1); delay(1000);
gpio_put(LEDE,1); gpio_put(LEDC,0); delay(1000);
gpio_put(LEDD,1); gpio_put(LEDA,1); gpio_put(LEDF,0); delay(1000);
gpio_put(LEDD,1); gpio_put(LEDA,1); gpio_put(LEDF,0); delay(1000);
gpio_put(LEDD,0); gpio_put(LEDA,0); gpio_put(LEDB,1); delay(1000);
gpio_put(LEDE,0); delay(1000);
N1(1); gpio_put(LEDA,0); delay(1000);
N8(1); delay(1000);
gpio_put(LEDE,1); delay(1000);
gpio_put(LEDE,0); gpio_put(LEDD,1); delay(1000);
gpio_put(LEDA,1); gpio_put(LEDB,1); gpio_put(LEDD,0); delay(1000);
gpio_put(LEDA,1); gpio_put(LEDB,1); gpio_put(LEDD,0); delay(1000);
gpio_put(LEDG,1); gpio_put(LEDC,1); gpio_put(LEDA,0); delay(1000);
gpio_put(LEDA,1); gpio_put(LEDF,1); gpio_put(LEDC,0); gpio_put(LEDB,0); gpio_put(LEDG,0); delay(1000);
gpio_put(LEDA,0); gpio_put(LEDB,1); gpio_put(LEDF,0); gpio_put(LEDC,1); delay(1000);
gpio_put(LEDD,1); delay(1000);
}
